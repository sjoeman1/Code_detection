n, x, y = map(int, input().split())
a = input().strip()

# Calculate the minimum cost
cost = 0
for i in range(len(a)):
    # Reverse a substring
    if a[i:i + 1] != '1':
        cost += x
    # Invert a substring
    if a[i:i + 1] != '0':
        cost += y

# Print the minimum cost
print(cost) if cost else print(0)
