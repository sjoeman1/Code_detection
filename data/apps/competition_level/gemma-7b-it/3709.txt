```python
n, k = map(int, input().split())

problems = []
for _ in range(n):
    problems.append(list(map(int, input().split())))

teams_problems = []
for i in range(k):
    teams_problems.append(set(problems[i]))

is_interesting = True
for team_problems in teams_problems:
    if len(team_problems.intersection(problems[:len(team_problems) // 2])) > len(team_problems) // 2:
        is_interesting = False

print("YES" if is_interesting else "NO")
```
