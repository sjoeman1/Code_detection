n, k, p = map(int, input().split())

a = list(map(int, input().split()))
b = list(map(int, input().split()))

a.sort()
b.sort()

# Calculate the minimum time needed to get to the office
time = 0
for i in range(len(a)):
    # Calculate the distance to the office
    distance = abs(b[0] - a[i])
    # Add the time needed to travel to the office
    time += distance

# Print the minimum time
print(time)
