```python
a, b, n, k = map(int, input().split())
s = input().split()

# Calculate the sum of the sequence
sum = 0
for i in range(len(s)):
    if s[i] == '-':
        sum -= 1
    else:
        sum += 1

# Calculate the remainder when the sum is divided by 10^{9} + 9
remainder = sum % (10 ** 9 + 9)

# Print the remainder
print(remainder)
```
