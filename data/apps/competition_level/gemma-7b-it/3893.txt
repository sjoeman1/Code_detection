```python
import numpy as np

# Read the input data
x1, y1 = map(int, input().split())
x2, y2 = map(int, input().split())
n = int(input())

# Store the road information
a = np.array(list(map(int, input().split())))
b = np.array(list(map(int, input().split())))
c = np.array(list(map(int, input().split())))

# Calculate the determinant of the matrix
det = np.linalg.det(np.array([[a[i], b[i], c[i]] for i in range(n)]))

# If the determinant is zero, it means that the university and home are on the same line, so the minimum number of steps is 1
if det == 0:
  print(1)

# Otherwise, the minimum number of steps is 2
else:
  print(2)
```
