```python
n, m = map(int, input().split())

grid = []
for _ in range(n):
    grid.append(input().split())

r, c = None, None
for r in range(n):
    for c in range(m):
        if grid[r][c].count('B') % 2 == 1:
            r_center = (r + 1) // 2
            c_center = (c + 1) // 2
            print(r_center, c_center)
            exit()

print("No solution")
```
