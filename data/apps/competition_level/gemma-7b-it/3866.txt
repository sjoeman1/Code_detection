```python
n = int(input())

if n == 0:
    print(-1)
else:
    a, b, c = [], [], []
    for _ in range(n):
        a.append(int(input().split()[0]))
        b.append(int(input().split()[0]))
        c.append(int(input().split()[0]))

    found = False
    for i in range(n):
        if a[i] + b[i] != c[i] % n:
            found = True

    if found:
        print(-1)
    else:
        print(*a)
        print(*b)
        print(*c)
```
