```python
n = int(input())
s = input()

# Calculate the length of the longest alternating subsequence
def longest_alternating_subsequence(s):
    n = len(s)
    dp = [-1] * n
    dp[0] = 1
    for i in range(1, n):
        if s[i] != s[i-1]:
            dp[i] = max(dp[i-1] + 1, dp[i])

    return dp[n-1]

# Find the maximum length of the longest alternating subsequence
max_length = longest_alternating_subsequence(s)

# Print the maximum length
print(max_length)
```
