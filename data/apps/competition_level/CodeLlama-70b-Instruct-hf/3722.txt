def answer(N, c_AA, c_AB, c_BA, c_BB):
    MOD = 10**9 + 7

    # Create a 2D array to store the number of possible strings with
    # the length of s being i and the last character being A or B
    dp = [[0, 0] for _ in range(N+1)]

    # Initialize the number of possible strings with the length of s
    # being 1 and the last character being A or B
    dp[1] = [1, 1]

    # Loop through the length of s from 2 to N
    for i in range(2, N+1):
        # For the last character being A
        if c_AA == 'A':
            dp[i][0] += dp[i-2][0]
        if c_AB == 'A':
            dp[i][0] += dp[i-2][1]

        # For the last character being B
        if c_BA == 'B':
            dp[i][1] += dp[i-2][0]
        if c_BB == 'B':
            dp[i][1] += dp[i-2][1]

        # Take modulo of the values to avoid overflow
        dp[i][0] %= MOD
        dp[i][1] %= MOD

    # Return the sum of possible strings with the length of s being N
    return (dp[N][0] + dp[N][1]) % MOD

# Read input from standard input
N = int(input())

# Call the answer function and print the result
print(answer(N, c_AA, c_AB, c_BA, c_BB))
