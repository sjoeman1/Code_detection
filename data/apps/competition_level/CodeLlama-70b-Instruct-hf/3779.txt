def astronaut(n, k, a):
    count = 0
    div_list = []
    for i in range(n):
        if a[i] % k == 0:
            div_list.append(a[i])
    for i in range(k):
        sum = 0
        for j in div_list:
            sum += j
            if sum % k == i:
                count += 1
                break
    return count
