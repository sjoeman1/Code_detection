    print('No solution')

# Otherwise, find the longest common substring of s and t.
else:
    # Initialize the longest common substring to an empty string.
    lcs = ''

    # Iterate through all possible substrings of s.
    for i in range(len(s)):

            if s[i:j] in t and len(s[i:j]) > len(lcs):
                lcs = s[i:j]

    # Check if the longest common substring is not found.
    if lcs == '':
        print('No solution')

    # Otherwise, rearrange the schedule according to the longest common substring.
    else:
        schedule = ''

        # Repeat the longest common substring until the number of 0s and 1s matches.
        while n0 > 0 or n1 > 0:
