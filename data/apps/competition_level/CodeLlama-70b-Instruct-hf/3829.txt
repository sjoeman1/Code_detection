def solve(m, n):
    # Initialize the answer as a variable to store the expected maximum number of dots
    answer = 0

    # Iterate over the faces of the dice from 1 to m
    for i in range(1, m + 1):
        # Calculate the probability of getting the current face
        probability = 1 / m

        # Calculate the expected maximum number of dots after tossing the dice n times
        expected_max = (i * (1 - (1 - probability) ** n))

        # Add the expected maximum to the answer
        answer += expected_max

    return answer


# Get the input from the user
