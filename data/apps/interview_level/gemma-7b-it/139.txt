```python
n, m = map(int, input().split())

graph = {i: set() for i in range(1, n + 1)}
for _ in range(m):
    u, v = map(int, input().split())
    graph[u].add(v)

acyclic = True
for v in graph:
    if v not in graph:
        continue
    for u in graph[v]:
        if u in graph[v]:
            acyclic = False

print("YES" if acyclic else "NO")
```
