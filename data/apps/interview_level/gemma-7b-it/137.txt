```python
n, p = map(int, input().split())
c = list(map(int, input().split()))

# Calculate the number of valid paths for a given parity
def valid_paths(n, p, c):
    dp = [-1] * (n + 1)
    dp[0] = 1
    for i in range(1, n + 1):
        dp[i] = dp[i - 1] * 2
        if c[i - 1] != -1 and c[i - 1] != c[i]:
            dp[i] += dp[i - 1]
    return dp[n]

# Calculate the answer
ans = valid_paths(n, p, c)

# Print the answer
print(ans % (10**9 + 7))
```
