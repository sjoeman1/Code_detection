```python

N = int(input())
A = []
for _ in range(N):
    A.append([int(input().split()[i]) for i in range(N)])

K = min(A[0])
R = min(A[0])
for i in range(1, N):
    K = min(K, abs(A[i][0] - A[0][0]) + abs(A[i][0] - A[i-1][0])
    R = min(R, abs(A[i][0] - A[0][0]) + abs(A[i][0] - A[i-1][0])

print(K + R)
```
