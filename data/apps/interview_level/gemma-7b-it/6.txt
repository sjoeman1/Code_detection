```python
t = int(input())

for _ in range(t):
    n, x = map(int, input().split())
    d = list(map(int, input().split()))
    h = list(map(int, input().split()))

    curX = x
    minBlows = float("inf")

    for blow in d:
        minHead = min(blow, curX)
        curX -= minHead
        curX += h[blow-1]

        if curX <= 0:
            minBlows = min(minBlows, blow)

    if minBlows != float("inf"):
        print(minBlows)
    else:
        print(-1)
```
