```python
d, k, a, b, t = map(int, input().split())

# Calculate the number of kilometers that Vasiliy can drive and the time it takes to drive those kilometers
km_driven = d // k
time_driven = km_driven * a

# Calculate the time it takes to repair the car and walk to the post office
time_repair = t
time_walk = (d - km_driven) * b

# Find the minimum time
time_min = time_driven + time_repair + time_walk

# Print the minimum time
print(time_min)
```
