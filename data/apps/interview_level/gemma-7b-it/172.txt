```python
n = int(input())
a = list(map(int, input().split()))
b = list(map(int, input().split()))

a.sort()
b.sort()

# Calculate the number of students with the same academic performance in both groups
a_dict = {}
for i in a:
    if i not in a_dict:
        a_dict[i] = 0
    a_dict[i] += 1

b_dict = {}
for i in b:
    if i not in b_dict:
        b_dict[i] = 0
    b_dict[i] += 1

# Calculate the minimum number of exchanges
min_exchanges = 0
for i in range(1, 6):
    if abs(a_dict[i] - b_dict[i]) > 0:
        min_exchanges += abs(a_dict[i] - b_dict[i])

# Print the minimum number of exchanges
if min_exchanges == 0:
    print(-1)
else:
    print(min_exchanges)
```
