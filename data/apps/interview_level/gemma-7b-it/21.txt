```python
n = int(input())
a = list(map(int, input().split()))

a.sort()

# Calculate the maximum distance between the minimum and maximum elements
max_distance = abs(a[0] - a[-1])

# Calculate the distance between the minimum and maximum elements after swapping one element
for i in range(n):
    a_copy = a.copy()
    a_copy[i] = a[0]
    a_copy.sort()
    min_distance = abs(a_copy[0] - a_copy[-1])
    max_distance = max(max_distance, min_distance)

# Print the maximum distance
print(max_distance)
```
