l_1, r_1, l_2, r_2, k = map(int, input().split())

# Calculate the overlapping minutes between l_1-r_1 and l_2-r_2
overlap = min(r_1, r_2) - max(l_1, l_2) + 1

# Subtract the minutes spent on printing
minutes_spent_on_printing = min(l_1, k) - max(l_1, k)

# Calculate the total number of minutes they can spend together
total_minutes = overlap - minutes_spent_on_printing

# Print the result
print(total_minutes)
