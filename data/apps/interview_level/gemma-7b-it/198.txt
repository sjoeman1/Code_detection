```python
n = int(input())

# Calculate the factors of n
factors = []
for i in range(1, int(n**0.5) + 1):
    if n % i == 0:
        factors.append(i)

# Count the number of ways to split the stick
num_ways = 0
for a, b, c in permutations(factors):
    if a + b + c == n and a != b and b != c:
        num_ways += 1

# Print the number of ways
print(num_ways)
```
