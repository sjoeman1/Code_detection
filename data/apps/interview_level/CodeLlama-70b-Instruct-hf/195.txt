```py
def count_students(A, B, C, N):
    # Check if a distribution of N students exists in which A students visited BugDonalds, B — BeaverKing, C — both of the restaurants and at least one student is left home (it is known that Vasya didn't pass the exam and stayed at home).
    # Output one integer — amount of students (including Vasya) who did not pass the exam.
    if N < A + B - C:
        return -1  # Impossible distribution
    else:
        return N - A - B + C + 1  # Count of students who didn't pass the exam, including Vasya


# Input
A, B, C, N = map(int, input().split())

# Output
print(count_students(A, B, C, N))

```
