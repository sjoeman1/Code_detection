```python
def min_moves(n, k):
    # Base cases
    if n == 2:
        return 6
    if k == 1 or k == n:
        return 2 * n - 3
    return 2 * n - 4


# Input
n, k = map(int, input().split())

# Output
print(min_moves(n, k))
```
