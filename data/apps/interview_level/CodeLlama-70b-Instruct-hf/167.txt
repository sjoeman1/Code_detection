def solve(a, b):
    # initialize the answer as an empty string
    ans = ""
    # initialize the index of the current position in string a
    i = 0
    # iterate through each character in string b
    for ch in b:
        # check if the character is in string a starting from the current index
        if ch in a[i:]:
            # if it is, then add it to the answer
            ans += ch
            # and update the current index in string a
            i = a.index(ch, i) + 1
    # return the answer
    return ans
