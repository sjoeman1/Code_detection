# Function to calculate the number of minutes they will be able to spend together
def number_of_minutes(l1, r1, l2, r2, k):
    # Check if Filya's visit overlaps with Sonya's sleep time
    if l2 > r1 or r2 < l1:
        return 0
    
    # Check if Sonya prinks during their visit
    if k >= l1 and k <= r1 and k >= l2 and k <= r2:
        return 0
    
    # Calculate the overlapping time
    overlap_start = max(l1, l2)
    overlap_end = min(r1, r2)
    overlap_time = overlap_end - overlap_start + 1
    
    # Remove the prink time if it falls within the overlapping time
    if k >= overlap_start and k <= overlap_end:
        overlap_time -= 1
    
    return overlap_time

# Read the input
l1, r1, l2, r2, k = map(int, input().split())

# Calculate the number of minutes they will be able to spend together
minutes = number_of_minutes(l1, r1, l2, r2, k)

# Print the result
print(minutes)

It first checks if Filya's visit overlaps with Sonya's sleep time, and if not, returns 0. If they do overlap, it checks if Sonya prinks during their visit, and if so, returns 0.
