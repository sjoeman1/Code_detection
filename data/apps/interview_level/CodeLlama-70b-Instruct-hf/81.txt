from math import gcd

def lcm(x, y):
    return (x * y) // gcd(x, y)

def solution(a, b):
    lcm_ab = lcm(a, b)
    lcm_ap_b = lcm(a + 1, b + 1)

    while lcm_ap_b < lcm_ab:
        a += 1
        b += 1
        lcm_ab = lcm_ap_b
        lcm_ap_b = lcm(a + 1, b + 1)

    return a - b

a, b = map(int, input().split())
k = solution(a, b)
print(k)
