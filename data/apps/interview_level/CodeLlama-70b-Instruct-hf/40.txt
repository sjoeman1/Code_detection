```python
# This function determines if a round is rated or not.
def is_rated(participants):
    # Check if any participant's rating has changed.
    for i in range(len(participants)):
        if participants[i][0] != participants[i][1]:
            return "rated"

    # Check if any participant with a lower rating took a better place.
    for i in range(1, len(participants)):
        if participants[i][0] < participants[i-1][0] and participants[i][1] > participants[i-1][1]:
            return "maybe"

    return "unrated"

# Read the number of participants.
n = int(input())

# Read the ratings for each participant.
participants = []
for _ in range(n):
    a, b = map(int, input().split())
    participants.append([a, b])

# Determine if the round is rated or not.
print(is_rated(participants))
```

This code takes input from the user, checks for each participant's ratings, and determines if the round is rated, unrated, or if it's impossible to determine.
