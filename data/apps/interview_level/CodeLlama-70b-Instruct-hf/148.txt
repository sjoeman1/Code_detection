```python
n, a, x, b, y = map(int, input().split())  # Read the input numbers

# Check if Daniel and Vlad will ever be at the same station
# at the same time during their journey

# Initialize the current station of Daniel and Vlad
current_station_daniel = a
current_station_vlad = b

# Initialize the number of minutes passed
minutes_passed = 0

# Keep looping until one of the trains leaves the subway
while current_station_daniel != x and current_station_vlad != y:
    # Increment the minutes passed
    minutes_passed += 1

    # Calculate the next station of Daniel and Vlad
    next_station_daniel = (current_station_daniel + 1) % n
    if next_station_daniel == 0:
        next_station_daniel = 1
    next_station_vlad = (current_station_vlad - 1) % n
    if next_station_vlad == 0:
        next_station_vlad = n

    # Check if Daniel and Vlad are at the same station
    if next_station_daniel == next_station_vlad:
        # If they are, print "YES" and break the loop
        print("YES")
        break
    else:
        # If they are not, update the current station of Daniel and Vlad
        current_station_daniel = next_station_daniel
        current_station_vlad = next_station_vlad

# If the loop is not broken, it means Daniel and Vlad are not at the same station
else:
    print("NO")
```
