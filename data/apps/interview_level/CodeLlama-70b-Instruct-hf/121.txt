position = []
for _ in range(4):

# Function to check if a given row, column, or diagonal is a winning line
def is_winning_line(line):
    if len(set(line)) == 1 and line[0] == 'x':
        return True
    return False

# Check if Ilya could win by making a single turn
for i in range(4):
    # Check if any row is a winning line
    if is_winning_line(position[i]):
        print("YES")
        break
    # Check if any column is a winning line
    if is_winning_line([position[j][i] for j in range(4)]):
        print("YES")
        break
    # Check if the main diagonal is a winning line
    if is_winning_line([position[j][j] for j in range(4)]):
        print("YES")
        break
    # Check if the secondary diagonal is a winning line
    if is_winning_line([position[j][3-j] for j in range(4)]):
        print("YES")
        break
else:
    # None of the winning lines were found, so Ilya cannot win by making a single turn
    print("NO")
