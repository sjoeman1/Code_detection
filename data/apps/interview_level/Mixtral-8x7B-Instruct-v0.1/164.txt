import math

def solve(y1, y2, yw, xb, yb, r):
    goal_width = y2 - y1
    dx = math.sqrt(r**2 - (yw - yb)**2)
    if dx > xb or dx < -xb:
        return -1
    xw = xb + dx
    if xw < 0 or xw > goal_width:
        return -1
    return xw
