def can_grasshopper_reach(n, k, line):
    start, target = line.index('G'), line.index('T')
    if (start + k) % n == target or (start - k) % n == target:
        return True
    return False

n, k = map(int, input().split())
line = input()
if can_grasshopper_reach(n, k, line):
    print("YES")
else:
    print("NO")
