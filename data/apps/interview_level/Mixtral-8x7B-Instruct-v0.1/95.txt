```python
arr = list(map(int, input().split()))

if n == 1:

        if arr[i] < arr[i + 1]:
            # increase
            j = i + 1

                j += 1
            if j == n - 1 or arr[j] > arr[j + 1]:
                i = j + 1
            else:
                flag = True
        elif arr[i] > arr[i + 1]:
            # decrease

                j += 1
            if j == n - 1 or arr[j] < arr[j + 1]:
                i = j + 1
            else:
                flag = True
        else:
            i += 1

    if flag:
        print("NO")
    else:
        print("YES")
```
