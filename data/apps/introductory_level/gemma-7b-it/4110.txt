```python
D, G = map(int, input().split())

p_list, c_list = [], []
for _ in range(D):
    p, c = map(int, input().split())
    p_list.append(p)
    c_list.append(c)

total_score = 0
for i in range(1, D + 1):
    total_score += p_list[i - 1] * i

perfect_bonus = 0
for i in range(1, D + 1):
    perfect_bonus += c_list[i - 1] if p_list[i - 1] == i else 0

required_problems = (total_score - perfect_bonus) // 100 + 1

print(required_problems)
```
