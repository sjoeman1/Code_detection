```python
n, k = map(int, input().split())
s = input().lower()

# Calculate the frequency of each character in the string
freq = {}
for i in s:
    if i not in freq:
        freq[i] = 0
    freq[i] += 1

# Calculate the minimum cost to obtain the set S of size k
cost = float('inf')
for i in range(1, k + 1):
    min_cost = float('inf')
    for j in freq:
        if freq[j] >= i:
            min_cost = min(min_cost, freq[j] * i - i)
    cost = min(cost, min_cost)

# If it is impossible to obtain the set S of size k, print -1
if cost == float('inf'):
    print(-1)
else:
    print(cost)
```
