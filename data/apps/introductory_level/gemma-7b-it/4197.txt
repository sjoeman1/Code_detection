```python3
N = int(input())

A = list(map(int, input().split()))

# Create a dictionary to store the number of students in the classroom at each time
students_in_class = {}

# Iterate over the A array and update the dictionary
for i in range(1, N + 1):
    students_in_class[i] = A[i - 1]

# Sort the students by the number of students in the classroom at each time
students_in_class_sorted = sorted(students_in_class.items(), key=lambda item: item[1])

# Print the student numbers in the order they entered the classroom
for student_number, num_students in students_in_class_sorted:
    print(student_number)
```
