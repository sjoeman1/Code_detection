```python
from math import pi

def calculate_area(r):
    return pi * r**2

def solve_question(r):
    area_circle_r = calculate_area(r)
    area_circle_1 = calculate_area(1)
    return int(area_circle_r / area_circle_1)

if __name__ == "__main__":
    r = int(input())
    answer = solve_question(r)
    print(answer)
```
