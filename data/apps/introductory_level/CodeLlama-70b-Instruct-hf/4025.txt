def solve(a, b, c):
    days = 0
    if a > 0:
        days += 3
    if b > 0:
        days += 2
    if c > 0:
        days += 1
    return days


if __name__ == '__main__':
    a, b, c = map(int, input().split())
    print(solve(a, b, c))
