import sys
from typing import List

def will_they_agree(n: int, m: int, x: int, y: int, x_cities: List[int], y_cities: List[int]) -> str:

    if x < y and x < Z <= y and all(xc < Z for xc in x_cities) and all(yc >= Z for yc in y_cities):
        return "No War"
    else:
        return "War"

def main():
    # Read input
    n, m, x, y = map(int, sys.stdin.readline().strip().split())
    x_cities = list(map(int, sys.stdin.readline().strip().split()))
    y_cities = list(map(int, sys.stdin.readline().strip().split()))

    # Check if war will break out
    result = will_they_agree(n, m, x, y, x_cities, y_cities)

    print(result)
