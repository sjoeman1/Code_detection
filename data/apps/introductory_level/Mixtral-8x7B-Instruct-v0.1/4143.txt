def min_time_needed(N, A, B, C, D, E):
    # Calculate the minimum number of vehicles needed for all people to travel
    num_vehicles = math.ceil(N/A)
    num_vehicles = min(num_vehicles, math.ceil(N/B))
    num_vehicles = min(num_vehicles, math.ceil(N/C))
    num_vehicles = min(num_vehicles, math.ceil(N/D))
    num_vehicles = min(num_vehicles, math.ceil(N/E))

    # Return the minimum time needed for all people to travel
