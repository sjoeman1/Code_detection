def solve(n, k):
    if (n & 1) or k < n:
        return "NO"

    cur_man, cur_woman = 1, 2
    res = [["man", "woman"] for _ in range(n)]
    for i in range(n):
        res[i][0] = cur_man
        res[i][1] = cur_woman
        if cur_man == k:
            cur_man = 1
            cur_woman += 1
        else:
            cur_man += 1

    if n > 1:
        if res[0][0] == res[1][0] or res[0][1] == res[1][1]:
            return "NO"

    for i in range(1, n - 1, 2):
        if res[i][0] == res[i + 1][0] or res[i][1] == res[i + 1][1]:
            return "NO"

    return "YES\n" + "\n".join([" ".join(map(str, x)) for x in res])

n, k = map(int, input().split())
print(solve(n, k))
